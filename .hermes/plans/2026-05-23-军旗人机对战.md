**Goal:** 创建一个终端命令行版的军旗人机对战游戏，AI 随机走棋（简单模式）。

## 游戏规则说明

### 棋子等级（从高到低）
1. 司令 (1)
2. 军长 (2)
3. 师长 (3)
4. 旅长 (4)
5. 团长 (5)
6. 营长 (6)
7. 连长 (7)
8. 排长 (8)
9. 工兵 (9)
10. 炸弹 (10) - 与任何棋子相遇同归于尽
11. 地雷 (11) - 固定放置，只能用工兵挖或炸弹炸
12. 军旗 (12) - 固定放置，被捉即输

### 特殊规则
- 工兵可以走铁路线（任意拐弯）
- 炸弹与任何棋子相遇同归于尽
- 地雷不能移动，只能被工兵挖或炸弹炸
- 军旗不能移动，被捉则游戏结束
- 普通棋子只能走到相邻格（上下左右斜，共8方向）
- 大本营不能放置棋子（开局时每个大本营放一个棋子）

# jiqi/piece.py
from enum import Enum

class PieceType(Enum):
    COMMANDER = "司令"
    ARMY_COMMANDER = "军长"
    DIVISION_COMMANDER = "师长"
    BRIGADE_COMMANDER = "旅长"
    REGIMENT_COMMANDER = "团长"
    BATTALION_COMMANDER = "营长"
    COMPANY_COMMANDER = "连长"
    PLATOON_LEADER = "排长"
    SAPPER = "工兵"
    BOMB = "炸弹"
    MINE = "地雷"
    FLAG = "军旗"

class Piece:
    RANK = {
        PieceType.COMMANDER: 1,
        PieceType.ARMY_COMMANDER: 2,
        PieceType.DIVISION_COMMANDER: 3,
        PieceType.BRIGADE_COMMANDER: 4,
        PieceType.REGIMENT_COMMANDER: 5,
        PieceType.BATTALION_COMMANDER: 6,
        PieceType.COMPANY_COMMANDER: 7,
        PieceType.PLATOON_LEADER: 8,
        PieceType.SAPPER: 9,
        PieceType.BOMB: 10,
        PieceType.MINE: 11,
        PieceType.FLAG: 12,
    }

    def __init__(self, piece_type: PieceType, side: str):
        self.piece_type = piece_type
        self.side = side
        self.is_alive = True
        self.is_mine = side == "human"

    @property
    def rank(self) -> int:
        return self.RANK[self.piece_type]

    @property
    def name(self) -> str:
        return self.piece_type.value

    def can_eat(self, target: "Piece") -> bool:
        if not target.is_alive:
            return False
        if self.piece_type == PieceType.FLAG:
            return False
        if self.piece_type == PieceType.MINE:
            return False
        if self.piece_type == PieceType.BOMB:
            return True
        if target.piece_type == PieceType.FLAG:
            return True
        if target.piece_type == PieceType.MINE:
            return self.piece_type == PieceType.SAPPER or self.piece_type == PieceType.BOMB
        return self.rank < target.rank
```

from jiqi.board import Board
from jiqi.piece import Piece, PieceType

def test_board_creation():
    board = Board()
    assert board.width == 7
    assert board.height == 11
    assert board.grid[0][0] is None

def test_place_and_get():
    board = Board()
    p = Piece(PieceType.COMMANDER, side="human")
    board.place(p, 0, 0)
    assert board.get(0, 0) == p

def test_place_on_occupied_fails():
    board = Board()
    p1 = Piece(PieceType.COMMANDER, side="human")
    p2 = Piece(PieceType.ARMY_COMMANDER, side="human")
    board.place(p1, 0, 0)
    assert not board.place(p2, 0, 0)

def test_remove():
    board = Board()
    p = Piece(PieceType.COMMANDER, side="human")
    board.place(p, 0, 0)
    board.remove(0, 0)
    assert board.get(0, 0) is None
```
